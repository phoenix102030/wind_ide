\section{Document Purpose}

This document outlines the current implementation of the IDE-Advection model. The goal is to clarify the following:
\begin{enumerate}[label=\arabic*.]
  \item Definition of state variables;
  \item Advection network output parameterization;
  \item Integration of covariance $\Sigma_t$ into the propagation shape;
  \item Allocation of parameters between the IDE and Machine Learning modules;
  \item Offline two-stage training and multi-step validation procedures.
\end{enumerate}

\section{State Variables and Observations}

We consider 3 spatial stations, each with two-dimensional wind components $(u, v)$. At time $t$, the state vector is defined as:
\begin{equation}
Y_t
=
\begin{bmatrix}
u_t(s_1) & v_t(s_1) & u_t(s_2) & v_t(s_2) & u_t(s_3) & v_t(s_3)
\end{bmatrix}^{\!\top}
\in \R^6.
\end{equation}

The observation vector is given by:
\begin{equation}
Z_t = Y_t + \varepsilon_t,
\qquad
\varepsilon_t \sim \Normal(0, R),
\end{equation}
where in the current implementation:
\begin{equation}
R = r_{\mathrm{obs}}^2 \Id_6.
\end{equation}
Thus, the observation equation is a linear Gaussian model where the observation matrix is the identity matrix.

\section{NWP-driven Advection Distribution}

\subsection{Definition of Random Advection}
The NWP encoder consumes the recent $L$ frames of gridded NWP inputs
\begin{equation}
X_{t-L+1:t} \in \R^{L \times C \times Y \times X},
\end{equation}
applies a frame-wise CNN, and then aggregates the temporal embeddings with a Transformer encoder.

Instead of outputting a free transition matrix directly, the network outputs the parameters of a paired advection distribution:
\begin{equation}
\begin{bmatrix}
A_{uu,t} \\
A_{vv,t}
\end{bmatrix}
\sim \Normal(\mu_t, \Sigma_t),
\qquad
\mu_t \in \R^4,\ \Sigma_t \in \R^{4\times 4},
\end{equation}
where
\begin{equation}
A_{uu,t}, A_{vv,t} \in \R^2,
\qquad
\mu_t =
\begin{bmatrix}
\mu_{uu,x}(t) & \mu_{uu,y}(t) & \mu_{vv,x}(t) & \mu_{vv,y}(t)
\end{bmatrix}^{\!\top}.
\end{equation}

\subsection{Network Output Parameterization}
The advection network outputs two objects:
\begin{enumerate}[label=\arabic*.]
  \item \textbf{Mean advection vector} $\mu_t \in \R^4$;
  \item \textbf{Joint covariance} $\Sigma_t \in \R^{4\times 4}$.
\end{enumerate}

The covariance matrix is constructed through a Cholesky factor:
\begin{equation}
L_t=
\begin{pmatrix}
\ell_{11} & 0 & 0 & 0\\
\ell_{21} & \ell_{22} & 0 & 0\\
\ell_{31} & \ell_{32} & \ell_{33} & 0\\
\ell_{41} & \ell_{42} & \ell_{43} & \ell_{44}
\end{pmatrix},
\qquad
\Sigma_t = L_t L_t^\top.
\end{equation}

\section{Spatial Transport Kernel}
\subsection{Theorem-inspired spatial propagation}

The physical transport part is defined on the three station locations. Let
\begin{equation}
h_{ij}=
\begin{bmatrix}
x_i-x_j \\
y_i-y_j
\end{bmatrix}
\in \R^2
\end{equation}
be the displacement from site $s_j$ to site $s_i$. This same spatial vector is used for all four component pairs $(u,u)$, $(u,v)$, $(v,u)$, $(v,v)$. In particular, when two components come from the same site, the spatial distance is zero even if one is $u$ and the other is $v$.

Following the corrected Lagrangian form in Theorem 2', we define one kernel for each target/source component pair. For $i,j \in \{u,v\}$:
\begin{equation}
K_t^{(ij)}(a,b)
\propto
\left|D_t^{(ij)}\right|^{-1/2}
\exp\left(
- {h_t^{(ij)}(a,b)}^\top
{D_t^{(ij)}}^{-1}
h_t^{(ij)}(a,b)
\right),
\end{equation}
\begin{equation}
h_t^{(uu)}(a,b)=h_{ab}-\Delta t\,\mu_{uu}(t),
\qquad
h_t^{(vv)}(a,b)=h_{ab}-\Delta t\,\mu_{vv}(t),
\end{equation}
\begin{equation}
h_t^{(uv)}(a,b)=h_{ab}-\Delta t\,(\mu_{uu}(t)-\mu_{vv}(t)),
\qquad
h_t^{(vu)}(a,b)=h_{ab}-\Delta t\,(\mu_{vv}(t)-\mu_{uu}(t)),
\end{equation}
and
\begin{equation}
D_t^{(uu)}=\Id_2+2\Delta t^2\Sigma_{uu,uu}(t)+\epsilon \Id_2,
\qquad
D_t^{(vv)}=\Id_2+2\Delta t^2\Sigma_{vv,vv}(t)+\epsilon \Id_2,
\end{equation}
\begin{equation}
D_t^{(uv)}=\Id_2+2\Delta t^2\left(
\Sigma_{uu,uu}(t)+\Sigma_{vv,vv}(t)-\Sigma_{uu,vv}(t)-\Sigma_{vv,uu}(t)
\right)+\epsilon \Id_2,
\end{equation}
\begin{equation}
D_t^{(vu)}=\Id_2+2\Delta t^2\left(
\Sigma_{vv,vv}(t)+\Sigma_{uu,uu}(t)-\Sigma_{vv,uu}(t)-\Sigma_{uu,vv}(t)
\right)+\epsilon \Id_2.
\end{equation}

\section{State Transition Equation}

The full transition is assembled directly from the four theorem-based kernels. Each block from source site $s_b$ to target site $s_a$ is
\begin{equation}
A_t^{(a,b)}=
\begin{pmatrix}
K_t^{(uu)}(a,b) & K_t^{(uv)}(a,b) \\
K_t^{(vu)}(a,b) & K_t^{(vv)}(a,b)
\end{pmatrix}.
\end{equation}

The state transition is
\begin{equation}
Y_{t+1} = A_t Y_t + \eta_t, \quad \eta_t \sim \Normal(0, Q_t),
\end{equation}
\begin{equation}
A_t = \rho_t \, \widetilde{A}_t, \qquad \rho_t = 1 - \Delta t\, \lambda(t),
\qquad
Q_t = q_{\mathrm{proc}}(t)^2 \Id_6,
\end{equation}
where $\widetilde{A}_t$ is the block matrix assembled from $\{A_t^{(i,j)}\}$ and $\lambda(t)$ is a time-varying damping parameter.

\section{Time-varying IDE Parameters}

The IDE parameters are no longer global constants. Instead, they are learned as time-varying sequences indexed by the absolute time location of each training window.

To avoid introducing a separate NWP-driven IDE network, the current code uses a piecewise-constant parameterization. Let the total sequence length be $T$, and let $w$ be a configurable temporal window size. Then each IDE parameter is first learned on a coarse knot grid:
\begin{equation}
\{\tilde q_k\}_{k=1}^{K},\quad
\{\tilde r_k\}_{k=1}^{K},\quad
\{\tilde p_k\}_{k=1}^{K},\quad
\{\tilde \lambda_k\}_{k=1}^{K},\quad
\{\tilde m_k\}_{k=1}^{K},
\end{equation}
with
\begin{equation}
K = \left\lceil \frac{T}{w} \right\rceil.
\end{equation}
Each knot value is then repeated within its window to produce the full time-varying series:
\begin{equation}
q_{\mathrm{proc}}(t), \quad
r_{\mathrm{obs}}(t), \quad
p_0(t), \quad
\lambda(t), \quad
m_0(t).
\end{equation}

This means:
\begin{itemize}
  \item $w=1$ gives the finest possible time resolution;
  \item larger $w$ gives a smoother, coarser parameter trajectory.
\end{itemize}

\section{Two-Stage Offline Training}

\subsection{Stage 1: Measurement-only IDE Fitting}
Stage 1 assumes $\mu_t=0$ and $\Sigma_t=0$, so the system reduces to a measurement-only IDE model. In this stage, the model fits the time-varying IDE parameter trajectories
\begin{equation}
q_{\mathrm{proc}}(t),\quad r_{\mathrm{obs}}(t),\quad p_0(t),\quad \lambda(t),\quad m_0(t)
\end{equation}
directly from measurement data.

\subsection{Stage 2: Alternating Training}
\begin{itemize}
  \item \textbf{Statistical Step}: Fix the advection network, update the time-varying IDE parameters $\theta_{\mathrm{ide}}$.
  \item \textbf{Advection Step}: Fix IDE parameters, update advection network parameters $\theta_{\mathrm{adv}}$ using a smoothness penalty on the 4-dimensional $\mu_t$ and the 4-by-4 $\Sigma_t$ trajectories.
\end{itemize}

The optimized parameter groups are therefore:
\begin{align}
\theta_{\mathrm{ide}}
&=
\left\{
\{\log \tilde q_k\}_{k=1}^{K},
\{\log \tilde r_k\}_{k=1}^{K},
\{\log \tilde p_k\}_{k=1}^{K},
\{\log \tilde \lambda_k\}_{k=1}^{K},
\{\tilde m_k\}_{k=1}^{K}
\right\}, \\
\theta_{\mathrm{adv}}
&=
\text{all trainable parameters in the NWP encoder, temporal encoder, mean head, and Cholesky head.}
\end{align}

\section{Multi-step Validation}

Validation uses a \textit{rolling forecast origin} approach. After filtering through the \texttt{history\_len}, the model predicts $H$ steps into the future using only NWP-generated $\mu_t$ and $\Sigma_t$, without feeding in future ground truth observations.
